```sql
CREATE TABLE public.ai_decisions (
  id BIGSERIAL PRIMARY KEY,
  device_id TEXT NOT NULL REFERENCES public.devices(id),
  appliance TEXT NOT NULL,
  icon TEXT,
  status TEXT NOT NULL DEFAULT 'on',        -- 'on' | 'off' | 'standby'

CREATE INDEX idx_ai_decisions_device ON public.ai_decisions (device_id, created_at DESC);

-- automation_rules: user-configured rules
CREATE TABLE public.automation_rules (
  id BIGSERIAL PRIMARY KEY,
  device_id TEXT REFERENCES public.devices(id),

  action TEXT NOT NULL,
  active BOOLEAN DEFAULT true,
  icon TEXT,
  icon_bg TEXT,
  icon_color TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- notifications: system event log
CREATE TABLE public.notifications (
  id BIGSERIAL PRIMARY KEY,
  device_id TEXT REFERENCES public.devices(id),
  icon TEXT,
  icon_bg TEXT,
  icon_color TEXT,
  title TEXT NOT NULL,
  description TEXT,
  unread BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

```sql

ALTER TABLE public.devices ADD COLUMN owner_id UUID REFERENCES auth.users(id);
CREATE INDEX idx_devices_owner ON public.devices (owner_id);

-- Update RLS: devices visible only to owner
DROP POLICY "dashboard can read devices" ON public.devices;
CREATE POLICY "owner can read devices" ON public.devices
  FOR SELECT TO authenticated USING (owner_id = (SELECT auth.uid()));

-- energy_readings: readable if device belongs to user
DROP POLICY "dashboard can read readings" ON public.energy_readings;
CREATE POLICY "owner can read readings" ON public.energy_readings
  FOR SELECT TO authenticated
  USING (device_id IN (SELECT id FROM public.devices WHERE owner_id = (SELECT auth.uid())));
```

### 6b. Create realtime hook

```typescript
// src/lib/use-realtime.ts
import { useEffect, useState } from "react";
import { insforge } from "./insforge";

export function useRealtime(table: string, filter?: string) {
  const [data, setData] = useState(null);
  useEffect(() => {
    const channel = insforge.realtime
      .subscribe(`${table}_changes`, { event: "*", schema: "public", table, filter })
      .on("postgres_changes", (payload) => {
        setData(payload.new);
      })
      .subscribe();
    return () => { insforge.realtime.unsubscribe(channel); };
  }, [table, filter]);
  return data;
}
```

## Phase 7: History Aggregation

**Goal:** Pre-compute hourly/daily summaries for fast analytics.

### 7a. Migration — summary tables

```sql
CREATE TABLE public.hourly_energy_summary (
  id BIGSERIAL PRIMARY KEY,
  device_id TEXT NOT NULL,
  hour_start TIMESTAMPTZ NOT NULL,
  avg_pv_power NUMERIC,
  avg_load_power NUMERIC,
  total_pv_kwh NUMERIC,
  total_load_kwh NUMERIC,
  reading_count INTEGER,
  UNIQUE(device_id, hour_start)
);

CREATE TABLE public.daily_energy_summary (
  id BIGSERIAL PRIMARY KEY,
  device_id TEXT NOT NULL,
  day_date DATE NOT NULL,
  total_pv_kwh NUMERIC,
  total_load_kwh NUMERIC,
  avg_battery_percent NUMERIC,
  peak_pv_power NUMERIC,
  peak_load_power NUMERIC,
  UNIQUE(device_id, day_date)
);
```

### 7b. Aggregation function + trigger

```sql
-- Trigger: after insert on energy_readings, update hourly summary
CREATE OR REPLACE FUNCTION public.update_hourly_summary()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.hourly_energy_summary (device_id, hour_start, avg_pv_power, avg_load_power, total_pv_kwh, total_load_kwh, reading_count)
  VALUES (NEW.device_id, date_trunc('hour', NEW.timestamp), NEW.pv_power, NEW.load_power,
          NEW.pv_power / 1000.0 * 5.0/3600.0, NEW.load_power / 1000.0 * 5.0/3600.0, 1)
  ON CONFLICT (device_id, hour_start) DO UPDATE SET
    avg_pv_power = (hourly_energy_summary.avg_pv_power * hourly_energy_summary.reading_count + NEW.pv_power) / (hourly_energy_summary.reading_count + 1),
    avg_load_power = (hourly_energy_summary.avg_load_power * hourly_energy_summary.reading_count + NEW.load_power) / (hourly_energy_summary.reading_count + 1),
    total_pv_kwh = hourly_energy_summary.total_pv_kwh + NEW.pv_power / 1000.0 * 5.0/3600.0,
    total_load_kwh = hourly_energy_summary.total_load_kwh + NEW.load_power / 1000.0 * 5.0/3600.0,
    reading_count = hourly_energy_summary.reading_count + 1;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER trg_update_hourly_summary
  AFTER INSERT ON public.energy_readings
  FOR EACH ROW EXECUTE FUNCTION public.update_hourly_summary();
```
